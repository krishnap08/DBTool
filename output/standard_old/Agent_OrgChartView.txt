SELECT ZZEmp.EmpId
  , ZZEmp.Emp
  , ZZEmp.FirstName
  , ZZEmp.LastName
  , ZZEmp.EmpEffDt AS OrgChartViewEffDt
  , ZZEmp.EmpEfdDt AS OrgChartViewEfdDt
  , ZZEml.Eml
  , ZZEml.EmpSts
  , ZZEml.Job
  , ZZEml.Pos
  , ZZEml.Sup
  , ZZEml.Dpt
  , ZZEml.Ent
  , ZZEml.IsPrm
FROM ZZEmp
  LEFT JOIN ZZEml ON ZZEmp.Emp = ZZEml.Emp AND ZZEml.EmlEfdDt IS NULL
